int number, count=0, sum=0, mul=1, rem, copy;
for(number = 1; number <= 100; number++){
copy = number;
while(copy > 0){
copy = copy / 10;
count++;
}
copy = number; cnt = count;
while(copy > 0){
rem = copy % 10;
while(cnt != 0){
mul = mul * rem;
cnt--;
}
sum = sum + mul;
copy = copy / 10;
mul = 1; cnt = count;
}
if(sum == number)
printf("It is a armstrong No.");
sum = 0; rem = 0; cnt = 0; count = 0
}
return 0;
}
